-- 💠 Tải Fluent UI (nếu có) — nếu không tải được, script vẫn tiếp tục
local Fluent
do
    local ok, res = pcall(function()
        return loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua", true))()
    end)
    if ok and res then
        Fluent = res
    else
        warn("Không tải được Fluent UI; GUI sẽ không hiển thị nhưng script vẫn chạy.")
        Fluent = nil
    end
end

-- Nếu Fluent có thì tạo cửa sổ, không thì tạo chức năng tối thiểu
local Window
if Fluent then
    Window = Fluent:CreateWindow({
        Title = "🌸 Cun Hub 🌸",
        SubTitle = "Tổng hợp script Roblox",
        Size = UDim2.fromOffset(450, 300),
        Acrylic = true,
        Theme = "Dark",
        MinimizeKey = Enum.KeyCode.End
    })
end
