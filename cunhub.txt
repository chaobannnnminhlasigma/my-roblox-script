if Window then
    Tabs.Scripts = Window:AddTab({Title = "Scripts"})
    Tabs.Social = Window:AddTab({Title = "TikTok"})
else
    -- Fake API: chỉ tạo object có AddButton làm no-op bảo toàn cú pháp
    local function fakeTab() 
        return { AddButton = function() return end, AddToggle = function() return end }
    end
    Tabs.Scripts = fakeTab()
    Tabs.Social = fakeTab()
end

-- Danh sách script mẫu
local scripts = {
    {title="LEMON", link="https://pastefy.app/MJw2J4T6/raw"},
    {title="CHILI HUB", link="https://raw.githubusercontent.com/tienkhanh1/spicy/main/Chilli.lua"},
    {title="ZZZ TOOL", link="https://pandadevelopment.net/virtual/file/570a5fa7765cbff8"}
}

for _,v in ipairs(scripts) do
    Tabs.Scripts:AddButton({
        Title = v.title,
        Description = "Chạy "..v.title,
        Callback = function()
            pcall(function()
                local code = game:HttpGet(v.link, true)
                if code and #code > 0 then
                    loadstring(code)()
                else
                    warn("Không tải được script:", v.title, v.link)
                end
            end)
        end
    })
end

Tabs.Social:AddButton({
    Title = "Follow TikTok 🌸",
    Description = "b.yt03",
    Callback = function()
        pcall(function() setclipboard("https://www.tiktok.com/@b.yt03") end)
        if Fluent then
            Fluent:Notify({Title="Đã sao chép!", Content="Dán link vào trình duyệt để mở TikTok.", Duration=3})
        end
    end
})
