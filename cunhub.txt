
-- 🌸 Cun Hub | Phiên bản GitHub rút gọn
repeat task.wait() until game:IsLoaded()

-- 🔘 Nút mở menu
local btn = Instance.new("ImageButton", game.CoreGui)
btn.Size, btn.Position = UDim2.new(0,40,0,40), UDim2.new(0.1,0,0.15,0)
btn.BackgroundColor3 = Color3.fromRGB(0,0,0)
btn.Draggable = true
btn.Image = "" -- 🖼️ thay bằng ID ảnh Roblox (nếu muốn)
Instance.new("UICorner", btn).CornerRadius = UDim.new(1,10)
btn.MouseButton1Click:Connect(function()
    game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.End, false, game)
end)

-- 💠 Giao diện chính (Fluent UI)
local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()
local Window = Fluent:CreateWindow({
    Title = "🌸 Cun Hub 🌸",
    SubTitle = "Tổng hợp script Roblox",
    Size = UDim2.fromOffset(450, 300),
    Acrylic = true,
    Theme = "Dark",
    MinimizeKey = Enum.KeyCode.End
})

-- 🧭 Tabs
local Tabs = {
    Scripts = Window:AddTab({Title = "Scripts"}),
    Social = Window:AddTab({Title = "TikTok"})
}

-- ⚙️ Danh sách script
local scripts = {
    {title="LEMON", link="https://pastefy.app/MJw2J4T6/raw"},
    {title="CHILI HUB", link="https://raw.githubusercontent.com/tienkhanh1/spicy/main/Chilli.lua"},
    {title="ZZZ TOOL", link="https://pandadevelopment.net/virtual/file/570a5fa7765cbff8"}
}

for _,v in ipairs(scripts) do
    Tabs.Scripts:AddButton({
        Title = v.title,
        Description = "Chạy "..v.title,
        Callback = function()
            loadstring(game:HttpGet(v.link))()
        end
    })
end

-- 💬 Liên hệ / TikTok
Tabs.Social:AddButton({
    Title = "Follow TikTok 🌸",
    Description = "b.yt03",
    Callback = function()
        setclipboard("https://www.tiktok.com/@b.yt03")
        Fluent:Notify({
            Title = "Đã sao chép!",
            Content = "Dán link vào trình duyệt để mở TikTok nhé!",
            Duration = 3
        })
    end
})
